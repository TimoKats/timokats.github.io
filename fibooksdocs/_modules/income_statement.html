    """
    This class contains the income statement. 
    The important objects are: content (contains the data),
    template (contains the excel template),
    standard_categories (contains categories like 'revenues' and 'expenses') and
    custom_categories (contains categories that are added by the user) 
    """
    def __init__(self, company_name):
        # settings
        template_file = 'functions/lib/income_statement_template.json'
        # meta data
        self.company_name = company_name
        # fibooks parts
        self.content = {}
        self.template = json.load(open(template_file))
        self.standard_categories = {'revenues':[],'expenses':[]}
        self.custom_categories = {}

[docs]    def set_standard_categories(self):
        current_catagory = 'revenues'
        for account in self.content:
            if account not in self.template['ignore']:
                self.standard_categories[current_catagory].append(account)
            if account in self.template['revenues']:
                current_catagory = 'expenses'
            elif account in self.template['expenses']: 
                current_catagory = 'other'

[docs]    def add_account(self, name, value, category=None):
        """
        Adds an account to content. The category is an optional parameter.
        For example, you can use this function to add 'gross profit $20' in 'revenues'.
        """
        if category is not None:
            self.custom_categories[category] = name
        self.content[name] = value



[docs]    def delete_account(self, name):
        """
        Deletes an account. Takes the account name as its only variable.
        """
        self.content.drop([name], axis=1, inplace=True)
        for category, accounts in self.standard_categories.items():
            for account in accounts:
                if account == name:
                    self.standard_categories[category].remove(account)
        for category in self.custom_categories:
            for account in category:
                if account == name:
                    self.standard_categories[category].remove(account)

[docs]    def load_excel(self, filename, min_row=None, max_row=None, min_column=None, max_column=None):
        """
        Loads a financial statement from an excel spreadsheet. Takes the filename as its only mandetory parameter.
        However, if the spreadsheet has a lot of 'noise' the user can also define the min/max rows/columns to extract from.
        """
        dataset = import_excel(filename, min_row, max_row, min_column, max_column)
        temp = {}
        for account, values in dataset.items():
            if account not in self.template['ignore']:
                temp[account] = values
        self.content = pd.DataFrame.from_dict(temp, orient='index').transpose()
        self.set_standard_categories()

[docs]    def get_total(self, category):
        """
        Returns the total value of a category (e.g. revenues). Takes category as its only parameter.
        Note, the category can be standard or one of the custom/added categories.
        """
        total = [0] * len(self.content.index)
        for column in self.content:
            try:
                if column in self.standard_categories[category]:
                    for index, cell in enumerate(self.content[column]):
                        total[index] += self.content[column][index]
            except:
                if column in self.custom_categories[category]:
                    for index, cell in enumerate(self.content[column]):
                        total[index] += self.content[column][index]                
        return total



[docs]    def get_netincome(self):
        """
        Returns the net income of the income statement
        """
        net_income = list()
        for rev, exp in zip(self.get_total('revenues'), self.get_total('expenses')):
            ni = rev - exp
            net_income.append(ni)
        return net_income

        """
        Resets the standard template for reading excel files. Takes a json file as parameter.
        Note, this json file should have the standard categories as fields (and ignore) and the last entry of a category as value.
        """
        income_statement_json = open(filename)
        self.template = json.load(income_statement_json)
