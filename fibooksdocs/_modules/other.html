[docs]def npv(cashflow, r=1):
    """
    Returns the net present value of a series of cashflows.
    Takes the r as an optional parameter.
    """
    return sum(f / ((1 + r) ** i) for i, f in enumerate(cashflow, 1))

[docs]def income_based_valuation(statement_of_cashflows, r=0):
    """
    Computes the value of a company based on the statement of cash flows.
    Note, it therefore uses the income based valuation method.
    """
    fcf = statement_of_cashflows.get_fcf()
    return sum(f / ((1 + r) ** i) for i, f in enumerate(fcf, 1))
